async function cekValidasi() { const bank = document.getElementById('bank').value; const account = document.getElementById('account').value; const btn = document.getElementById('btnCek'); const loader = document.getElementById('loading'); const result = document.getElementById('result'); if(!account) return alert("Nomor rekening wajib diisi!"); btn.classList.add('hidden'); loader.classList.remove('hidden'); result.classList.add('hidden'); try { // MANGGIL BACKEND VERCEL KAMU SENDIRI const response = await fetch(`/api/check?bank=${bank}&account=${account}`); const data = await response.json(); if (data.success) { document.getElementById('owner').innerText = data.nama; document.getElementById('aiTip').innerText = "AI Security: " + data.aiText; result.classList.remove('hidden'); } else { alert("Gagal: " + (data.message || "Rekening tidak ditemukan.")); } } catch (error) { alert("Terjadi kesalahan sistem. Coba lagi."); } finally { btn.classList.remove('hidden'); loader.classList.add('hidden'); } }
