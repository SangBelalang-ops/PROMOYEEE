async function cekValidasi() { const bank = document.getElementById('bank').value; const account = document.getElementById('account').value; // UI Loading... document.getElementById('btnCek').classList.add('hidden'); document.getElementById('loading').classList.remove('hidden'); try { // PANGGIL BACKEND VERCEL KAMU SENDIRI // Bukan panggil api.co.id secara langsung const response = await fetch(`/api/check?bank=${bank}&account=${account}`); const result = await response.json(); if (result.success) { document.getElementById('owner').innerText = result.nama; document.getElementById('aiTip').innerText = "AI Security: " + result.aiText; document.getElementById('result').classList.remove('hidden'); } else { alert("Data tidak ditemukan!"); } } catch (err) { alert("Terjadi kesalahan sistem."); } finally { document.getElementById('btnCek').classList.remove('hidden'); document.getElementById('loading').classList.add('hidden'); } }
